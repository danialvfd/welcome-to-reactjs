import React from 'react';
import { BrowserRouter as Router, Route, Switch, Link } from 'react-router-dom';
import ClassComponent from './ClassComponent';
import FunctionalComponent from './FunctionalComponent';

const App = () => {
    return (
        <Router>
            <div>
                <nav>
                    <ul>
                        <li>
                            <Link to="/">Page 1 (Functional)</Link>
                        </li>
                        <li>
                            <Link to="/page2">Page 2 (Class)</Link>
                        </li>
                    </ul>
                </nav>

                <Switch>
                    <Route path="/" exact component={FunctionalComponent} />
                    <Route path="/page2" component={ClassComponent} />
                </Switch>
            </div>
        </Router>
    );
};

export default App;
